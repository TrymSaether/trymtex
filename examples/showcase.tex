\section{Algorithms}

\begin{algorithm}[htbp]
  \caption{Quicksort}
  \label{alg:quicksort}

  \KwIn{Array $A[1..n]$, indices $p,r$}
  \KwOut{Sorted subarray $A[p..r]$}

  \SetKwFunction{Quicksort}{Quicksort}
  \SetKwFunction{Partition}{Partition}

  \BlankLine
  \uIf{$p < r$}{
    $q \gets \Partition{A, p, r}$\;
    \Quicksort{A, p, q-1}\;
    \Quicksort{A, q+1, r}\;
  }
\end{algorithm}

\begin{algorithm}[htbp]
  \caption{Partition}
  \label{alg:partition}

  \KwIn{Array $A[1..n]$, indices $p,r$}
  \KwOut{Partition index $q$}

  \SetKwFunction{Partition}{Partition}

  \BlankLine
  $x \gets A[r]$\tcp*{pivot}
  $i \gets p - 1$\;
  \For{$j \gets p$ \KwTo $r-1$}{
    \If{$A[j] \le x$}{
      $i \gets i + 1$\;
      swap $A[i]$ and $A[j]$\;
    }
  }
  swap $A[i+1]$ and $A[r]$\;
  \KwRet{$i+1$}\;
\end{algorithm}
