\begin{definition}{Norm}{norm}
  A norm on a vector space \(V\) is a map \( \norm{\cdot} : V \to \RR \).
\end{definition}

\begin{lemma}{Nonnegativity}{nonneg}
  For all \( x \in \RR \), \( x^2 \ge 0 \).
\end{lemma}

\begin{theorem}{Pythagorean theorem}{pythagorean}
  For a right triangle with legs \(a\) and \(b\) and hypotenuse \(c\), we have \( a^2 + b^2 = c^2 \).
\end{theorem}

\begin{problem}{Right triangles}{right-triangles}
  Find the hypotenuse of a right triangle with legs \(3\) and \(4\).
\end{problem}

\begin{exercise}{Squares}{squares}
  Show that \( x^2 \ge 0 \) for every real number \(x\).
\end{exercise}

\begin{solution}{Squares}{squares}
  If \(x \ge 0\), then \(x^2 \ge 0\). If \(x < 0\), then \(-x > 0\) and \(x^2 = (-x)^2 \ge 0\).
\end{solution}
