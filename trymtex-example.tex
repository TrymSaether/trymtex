\section{A theorem}

\begin{definition*}[Norm]{def:norm}
  Let $V$ be a vector space. A function $\norm{\cdot}\colon V\to \R$ is a norm if it
  satisfies positivity, homogeneity, and the triangle inequality.
\end{definition*}

\begin{theorem*}[Cauchy--Schwarz]{thm:cs}
  For $x,y\in\R^n$,
  \[
    \abs{\inner{x,y}} \le \norm{x}\norm{y}.
  \]
\end{theorem*}

\begin{proof}
  This follows from the non-negativity of $\norm{x-\lambda y}^2$ with a suitable choice
  of $\lambda$.
\end{proof}

\begin{example*}[Derivative]{ex:dd}
  We write
  \[
    \dd{t} t^2 = 2t,
    \qquad
    \ddn{2}{t} t^3 = 6t.
  \]
\end{example*}
