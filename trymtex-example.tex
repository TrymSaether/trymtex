Here is some mathematics:
\[
  \R \subseteq \C,\qquad \norm{x} = \sqrt{\inner{x,x}}.
\]

\begin{definition}{Metric space}{metric-space}
  A metric space is a pair $(X,d)$ where $d$ is a distance function on $X$.
\end{definition}

\begin{theorem}{Triangle inequality}{triangle}
  For all $x,y \in \R^n$,
  \[
    \norm{x+y} \le \norm{x} + \norm{y}.
  \]
\end{theorem}

\begin{proof}
  Standard.
\end{proof}

\begin{algorithm}[H]
  \KwInput{A vector $x \in \R^n$}
  \KwOutput{$\norm{x}$}
  \Return{$\sqrt{\inner{x,x}}$}
  \caption{Euclidean norm}
\end{algorithm}
