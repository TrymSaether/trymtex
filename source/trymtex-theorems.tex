\ProvidesFile{trymtex-theorems.tex}[2026/04/09 Theorem module]

\tcbset{
  trymtex/theorem/common/.style={
    enhanced,
    breakable,
    before skip=10pt,
    after skip=10pt,
    separator sign=.,
    label separator={},
    top=4pt,
    bottom=4pt,
    left=6pt,
    right=6pt
  },
  trymtex/theorem/framed/.style 1 args={
    trymtex/theorem/common,
    enhanced jigsaw,
    colback=#1!10,
    colframe=#1!80!black,
    fonttitle=\bfseries
  },
  trymtex/theorem/ruled/.style 1 args={
    trymtex/theorem/common,
    enhanced jigsaw,
    colback=white,
    colframe=white,
    frame hidden,
    boxrule=0pt,
    borderline west={2pt}{0pt}{#1},
    left=8pt,
    right=4pt,
    top=2pt,
    bottom=2pt,
    fonttitle=\bfseries,
    coltitle=black,
    attach title to upper,
    after title={.\ },
    description font=\normalfont,
    description delimiters={(}{)}
  }
}

\newtcbtheorem{definition}{Definition}{
  trymtex/theorem/framed=def-color,
  top=3pt,
  bottom=3pt,
  left=3pt,
  right=3pt
}{def}

\newtcbtheorem{theorem}{Theorem}{
  trymtex/theorem/framed=thm-color,
  coltitle=black,
  attach boxed title to top left={xshift=10pt,yshift=-\tcboxedtitleheight/2},
  boxed title style={
    colback=thm-color!10,
    colframe=thm-color!80!black,
    height=16pt,
    bean arc
  },
  top=6pt,
  bottom=3pt,
  left=3pt,
  right=3pt
}{thm}

\newtcbtheorem{lemma}{Lemma}{
  trymtex/theorem/common,
  enhanced jigsaw,
  colback=lem-color!10,
  colframe=lem-color!80!black,
  boxrule=0pt,
  fonttitle=\bfseries,
  attach boxed title to top left={yshift=-\tcboxedtitleheight},
  boxed title style={
    boxrule=0pt,
    boxsep=2pt,
    colback=lem-color!80!black,
    colframe=lem-color!80!black,
    interior code={
      \fill[lem-color!80!black]
        (interior.north west) --
        (interior.south west) --
        ([xshift=-2mm]interior.south east) --
        ([xshift=2mm]interior.north east) -- cycle;
    }
  },
  frame hidden,
  borderline north={1pt}{0pt}{lem-color!80!black},
  before upper={\hspace{\tcboxedtitlewidth}},
  top=4pt,
  bottom=3pt,
  left=5pt,
  right=5pt
}{lem}

\newtcbtheorem{example}{Example}{
  trymtex/theorem/ruled=exm-color,
  coltitle=exm-color
}{exm}

\newtcbtheorem{remark}{Remark}{
  trymtex/theorem/ruled=rem-color
}{rem}

\newtcbtheorem{proposition}{Proposition}{
  trymtex/theorem/framed=prop-color,
  attach boxed title to top left={yshift=-\tcboxedtitleheight},
  boxed title style={
    boxrule=0pt,
    boxsep=2.5pt,
    colback=prop-color!80!black,
    colframe=prop-color!80!black,
    sharp corners=uphill
  },
  top=\tcboxedtitleheight,
  bottom=3pt,
  left=3pt,
  right=3pt
}{prop}

\newtcbtheorem{corollary}{Corollary}{
  trymtex/theorem/ruled=cor-color
}{cor}

\newtcbtheorem{pitfall}{Pitfall}{
  trymtex/theorem/ruled=pitfall-color
}{pit}

\RenewDocumentEnvironment{proof}{O{\proofname}}{%
  \begin{tcolorbox}[
    blanker,
    breakable,
    borderline west={2pt}{0pt}{thm-color},
    before skip=10pt,
    after skip=10pt,
    left=8pt,
    right=0pt,
    top=2pt,
    bottom=2pt
  ]
  \textbf{#1}.\enspace
}{%
  \end{tcolorbox}
}
