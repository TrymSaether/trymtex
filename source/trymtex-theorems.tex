\tcbset{
  trymtex/theorem/common/.style={
      enhanced,
      breakable,
      before skip=10pt,
      after skip=10pt,
      separator sign=.,
      label separator={:},
      top=2mm,
      bottom=2mm,
      left=4mm,
      right=4mm,
    },
  trymtex/theorem/framed/.style={
      trymtex/theorem/common,
      enhanced jigsaw,
      colback=#1!10,
      colframe=#1!80!base-black,
      fonttitle=\bfseries
    },
  trymtex/theorem/ruled/.style={
      trymtex/theorem/common,
      enhanced jigsaw,
      colback=base-gray,
      colframe=base-gray,
      frame hidden,
      boxrule=0pt,
      fonttitle=\bfseries,
      coltitle=base-black,
      attach title to upper,
      after title={.\ },
      description font=\normalfont,
      description delimiters={(}{)}
    },
  trymtex/theorem/ruled/solid/.style={
      trymtex/theorem/ruled,
      borderline west={2pt}{0pt}{#1},
    },
  trymtex/theorem/ruled/dashed/.style={
      trymtex/theorem/ruled,
      borderline west={2pt}{0pt}{#1,dashed},
    },
  trymtex/theorem/study/.style={
      trymtex/theorem/common,
      enhanced jigsaw,
      colback=#1!4,
      colframe=base-gray,
      frame hidden,
      boxrule=0pt,
      borderline west={2pt}{0pt}{#1!70!base-black},
      fonttitle=\bfseries,
      coltitle=base-black,
      attach title to upper,
      after title={.\ },
      description font=\normalfont,
      description delimiters={(}{)}
    }
}

\newtcbtheorem{definition}{Definition}{
  trymtex/theorem/framed=math-definition,
}{def}

\newtcbtheorem{theorem}{Theorem}{
  trymtex/theorem/framed=math-theorem,
  coltitle=base-black,
  attach boxed title to top left={xshift=10pt,yshift=-\tcboxedtitleheight/2},
  boxed title style={
      colback=math-theorem!10,
      colframe=math-theorem!80!base-black,
      height=16pt,
      bean arc
    },
}{thm}

\newtcbtheorem{lemma}{Lemma}{
  trymtex/theorem/common,
  enhanced jigsaw,
  colback=math-lemma!10,
  colframe=math-lemma!80!base-black,
  boxrule=0pt,
  fonttitle=\bfseries,
  attach boxed title to top left={yshift=-\tcboxedtitleheight},
  boxed title style={
      boxrule=0pt,
      boxsep=2pt,
      colback=math-lemma!80!base-black,
      colframe=math-lemma!80!base-black,
      interior code={
          \fill[math-lemma!80!base-black]
          (interior.north west) --
          (interior.south west) --
          ([xshift=-2mm]interior.south east) --
            ([xshift=2mm]interior.north east) -- cycle;
        }
    },
  frame hidden,
  borderline north={1pt}{0pt}{math-lemma!80!base-black},
  before upper={\hspace{\tcboxedtitlewidth}},
}{lem}
