\newtcolorbox{filingbox}[2][]{
  enhanced jigsaw,
  breakable,
  colback=surface-filing!10,
  colframe=surface-filing!80!base-black,
  fonttitle=\bfseries,
  coltitle=base-black,
  attach boxed title to top left={xshift=10pt,yshift=-\tcboxedtitleheight/2},
  boxed title style={
    colback=surface-filing!10,
    colframe=surface-filing!80!base-black,
    height=16pt,
    bean arc
  },
  title={#2},
  before skip=10pt,
  after skip=10pt,
  top=6pt,
  bottom=2pt,
  left=2pt,
  right=2pt,
  #1
}
