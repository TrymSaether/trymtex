\ProvidesFile{trymtex-boxes.tex}[2026/04/09 Utility box module]

\tcbset{
  trymtex/box/base/.style={
      enhanced jigsaw,
      breakable,
      fonttitle=\bfseries,
      before skip=10pt,
      after skip=10pt,
      top=2pt,
      bottom=2pt,
      left=2pt,
      right=2pt
    },
  trymtex/box/framed-title/.style={
      trymtex/box/base,
      colback=#1!10,
      colframe=#1!80!base-black,
      coltitle=base-black,
      attach boxed title to top left={xshift=10pt,yshift=-\tcboxedtitleheight/2},
      boxed title style={
          colback=#1!10,
          colframe=#1!80!base-black,
          height=16pt,
          bean arc
        },
      top=6pt
    },
  trymtex/box/slanted-title/.style={
      trymtex/box/base,
      colback=#1!10,
      colframe=#1!80!base-black,
      attach boxed title to top left={yshift=-\tcboxedtitleheight},
      boxed title style={
          boxrule=0pt,
          boxsep=2.5pt,
          colback=#1!80!base-black,
          colframe=#1!80!base-black,
          sharp corners=uphill
        },
      top=\tcboxedtitleheight
    },
  trymtex/box/flag/.style={
      trymtex/box/base,
      colback=#1!10,
      colframe=#1!80!base-black,
      coltitle=base-black,
      boxrule=0pt,
      attach title to upper,
      frame hidden,
      borderline west={2pt}{0pt}{#1},
      sharp corners,
      left=5pt,
      right=5pt
    }
}

\newtcolorbox{filingbox}[2][]{
  trymtex/box/framed-title=surface-filing,
  title={#2},
  #1
}

\newtcolorbox{railingbox}[2][]{
  trymtex/box/slanted-title=surface-railing,
  title={#2},
  #1
}

\newtcolorbox{flagbox}[2][]{
  trymtex/box/flag=surface-flag,
  title={#2},
  #1
}
