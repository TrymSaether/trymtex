\ProvidesFile{trymtex-fonts.tex}[2026/04/09 Font setup module]

\newcommand{\trymtex@setup@pdftex@lmodern}{%
  \RequirePackage{lmodern}
}

\newcommand{\trymtex@setup@pdftex@pagella}{%
  \RequirePackage{newpxtext}
  \RequirePackage{newpxmath}
}

\newcommand{\trymtex@setup@pdftex@ibmplex}{%
  \RequirePackage{lmodern}
  \trymtex@warning{IBM Plex is not supported under pdfTeX; using lmodern instead}%
}

\newcommand{\trymtex@setup@pdftex@fira}{%
  \RequirePackage{lmodern}
  \trymtex@warning{Fira is not supported under pdfTeX; using lmodern instead}%
}

\newcommand{\trymtex@setup@pdftex@none}{}

\newcommand{\trymtex@setup@unicode@base}{%
  \RequirePackage{fontspec}
  \RequirePackage[
    math-style=ISO,
    warnings-off={mathtools-colon,mathtools-overbracket}
  ]{unicode-math}
  \providecommand{\bm}[1]{\symbf{##1}}%
}

\newcommand{\trymtex@setup@unicode@lmodern}{%
  \trymtex@setup@unicode@base
  \setmainfont{Latin Modern Roman}
  \setsansfont{Latin Modern Sans}
  \setmonofont{Latin Modern Mono}
  \setmathfont{Latin Modern Math}
}

\newcommand{\trymtex@setup@unicode@pagella}{%
  \trymtex@setup@unicode@base
  \setmainfont{TeX Gyre Pagella}
  \setsansfont{TeX Gyre Heros}
  \setmonofont{Inconsolata}
  \setmathfont{TeX Gyre Pagella Math}
}

\newcommand{\trymtex@setup@unicode@ibmplex}{%
  \trymtex@setup@unicode@base
  \setmainfont{IBM Plex Serif}
  \setsansfont{IBM Plex Sans}
  \setmonofont{IBM Plex Mono}
  \setmathfont{STIX Two Math}
}

\newcommand{\trymtex@setup@unicode@fira}{%
  \trymtex@setup@unicode@base
  \setmainfont{Fira Sans}[
    BoldFont   = {Fira Sans Bold},
    ItalicFont = {Fira Sans Italic}
  ]
  \setsansfont{IBM Plex Sans Condensed}[Scale=0.95]
  \setmonofont{Fira Code}[
    Scale=0.9,
    Contextuals=Alternate
  ]
  \setmathfont{Fira Math}
  \setmathfont{STIX Two Math}[
    range={
        \mathscr,\mathbb,\mathfrak,\mathcal,
        \setminus,\Re,\Im,\triangleright,\top,
        \ddots,\vdots,\star,\bigcup,\succeq
    },
    Scale=MatchUppercase
  ]
}

\newcommand{\trymtex@setup@unicode@none}{%
  \trymtex@setup@unicode@base
}

\newcommand{\trymtex@setupfonts}{%
  \ifPDFTeX
    \RequirePackage{amssymb}
    \RequirePackage{bm}
    \trymtex@use@setup{pdftex}{\trymtex@fontset}
  \else
    \trymtex@use@setup{unicode}{\trymtex@fontset}
  \fi
}
